SELECT * FROM ORDER_CUSTOMER_DATE_PRICE;

CREATE TABLE ORDER_CUSTOMER_DATE_PRICE (CUSTOMER_NAME VARCHAR(25), ORDER_DATE DATE, ORDER_TOTAL_PRICE NUMBER(12,2), BATCH_ID FLOAT);

SELECT C.C_NAME AS CUSTOMER_NAME, O.O_ORDERDATE AS ORDER_DATE, SUM(O.O_TOTALPRICE) AS ORDER_TOTAL_PRICE
FROM ORDERS_RAW O JOIN CUSTOMER_RAW C ON O.O_CUSTKEY = C.C_CUSTKEY
WHERE O_ORDERSTATUS= 'F'
GROUP BY C_NAME,O_ORDERDATE
ORDER BY O_ORDERDATE
